//Table SQL

CREATE TABLE notes (
	id INT(11) NOT NULL AUTO_INCREMENT,
    note_title VARCHAR(255) NOT NULL,
	note_description TEXT,
    date_created DATETIME NOT NULL DEFAULT  CURRENT_TIME,
    PRIMARY KEY (id)
);
